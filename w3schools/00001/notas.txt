/* align-content no toma valores de izquierda y derecha */

align-content: center;
/* Empaque los artículos alrededor del centror */

align-content: start;
/* Reconocer artículos desde el principio */

align-content: end;
/* Empacar artículos desde el final */

align-content: flex-start;
/* Empaque artículos flexibles desde el principio */

align-content: flex-end;
/* Empaque los artículos flexibles desde el final */



/* Alineación normal */
align-content: normal;


/* Baseline alignment */
align-content: baseline;
align-content: first baseline;

/* Alineación distribuida */
align-content: space-between;
/* Distribute items evenly The first item is flush with the start, the last is flush with the end */

align-content: space-around;
/* Distribute items evenly Items have a half-size space on either end */

align-content: space-evenly;
/* Distribute items evenly Items have equal space around them */

align-content: stretch;
/* Distribute items evenly Stretch 'auto'-sized items to fit the container */

align-content: safe center;

align-content: inherit;
